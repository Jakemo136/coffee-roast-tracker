# UI & Features Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the full UI and feature set defined in the 2026-03-30 design spec — 9 vertical slices delivering design tokens, dashboard, roast detail, flavor system, upload modal, bean pages, scraping, comparison view, settings, and shared view.

**Architecture:** Vertical slices — each slice delivers schema changes + GraphQL + UI + tests as one unit. Shared components (StarRating, FlavorPill, Modal) emerge in the first slice that needs them and are reused. Server follows the existing resolver → service pattern. Client uses CSS Modules, gql.tada, and Apollo Client 4 with `useSuspenseQuery`.

**Tech Stack:** React 19 + TypeScript + Vite + Apollo Client 4 (gql.tada) | Node.js + Apollo Server 5 + Prisma 7 + PostgreSQL | Chart.js + react-chartjs-2 | Vitest + RTL + MSW (client) | Jest + supertest (server)

```ts
import { celsiusToFahrenheit } from "./tempConversion";

type TempUnit = "CELSIUS" | "FAHRENHEIT";

export function formatDuration(seconds: number | null | undefined): string {
  if (seconds == null) return "—";
  const mins = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${mins}:${secs.toString().padStart(2, "0")}`;
}

export function formatTemp(
  celsius: number | null | undefined,
  unit: TempUnit,
): string {
  if (celsius == null) return "—";
  if (unit === "FAHRENHEIT") {
    return `${Math.round(celsiusToFahrenheit(celsius))}°F`;
  }
  return `${Math.round(celsius)}°C`;
}

export function formatDate(iso: string | null | undefined): string {
  if (!iso) return "—";
  const d = new Date(iso);
  return d.toLocaleDateString("en-US", { month: "short", day: "numeric" });
}
```

```ts
import type { PrismaClient } from "@prisma/client";
import { GraphQLError } from "graphql";

const CATEGORY_COLORS: Record<string, string> = {
  FLORAL: "#c27a8a",
  HONEY: "#c9a84c",
  SUGARS: "#bda67a",
  CARAMEL: "#a88545",
  FRUITS: "#d45f5f",
  CITRUS: "#b8b44f",
  BERRY: "#7a4a6e",
  COCOA: "#8b5e4b",
  NUTS: "#8a7a4a",
  RUSTIC: "#6b6b4a",
  SPICE: "#a07050",
  BODY: "#5a4a3a",
  OFF_FLAVOR: "#c44a3b",
};

export class FlavorService {
  constructor(private prisma: PrismaClient) {}

  async flavorDescriptors(isOffFlavor?: boolean) {
    const where = isOffFlavor != null ? { isOffFlavor } : {};
    return this.prisma.flavorDescriptor.findMany({
      where,
      orderBy: [{ category: "asc" }, { name: "asc" }],
    });
  }

  async createFlavorDescriptor(name: string, category: string) {
    const color = CATEGORY_COLORS[category];
    if (!color) {
      throw new GraphQLError(`Invalid category: ${category}`, {
        extensions: { code: "BAD_USER_INPUT" },
      });
    }

    return this.prisma.flavorDescriptor.create({
      data: {
        name,
        category: category as any,
        isOffFlavor: category === "OFF_FLAVOR",
        isCustom: true,
        color,
      },
    });
  }

  async setRoastFlavors(userId: string, roastId: string, descriptorIds: string[]) {
    // Verify roast ownership
    const roast = await this.prisma.roast.findFirst({
      where: { id: roastId, userId },
    });
    if (!roast) {
      throw new GraphQLError("Roast not found", {
        extensions: { code: "NOT_FOUND" },
      });
    }

    // Delete existing positive flavors, then insert new ones
    await this.prisma.$transaction(async (tx) => {
      await tx.roastFlavor.deleteMany({
        where: {
          roastId,
          descriptor: { isOffFlavor: false },
        },
      });

      if (descriptorIds.length > 0) {
        await tx.roastFlavor.createMany({
          data: descriptorIds.map((descriptorId) => ({
            roastId,
            descriptorId,
          })),
        });
      }
    });

    return this.prisma.roast.findUniqueOrThrow({
      where: { id: roastId },
      include: {
        bean: true,
        roastFiles: true,
        roastProfile: true,
        roastFlavors: { include: { descriptor: true } },
      },
    });
  }

  async setRoastOffFlavors(userId: string, roastId: string, descriptorIds: string[]) {
    const roast = await this.prisma.roast.findFirst({
      where: { id: roastId, userId },
    });
    if (!roast) {
      throw new GraphQLError("Roast not found", {
        extensions: { code: "NOT_FOUND" },
      });
    }

    await this.prisma.$transaction(async (tx) => {
      await tx.roastFlavor.deleteMany({
        where: {
          roastId,
          descriptor: { isOffFlavor: true },
        },
      });

      if (descriptorIds.length > 0) {
        await tx.roastFlavor.createMany({
          data: descriptorIds.map((descriptorId) => ({
            roastId,
            descriptorId,
          })),

      include: {
        bean: true,
        roastFiles: true,
        roastProfile: true,
        roastFlavors: { include: { descriptor: true } },
```

```tsx
import styles from "./StarRating.module.css";

interface StarRatingProps {
  value: number | null;
  onChange?: (rating: number) => void;
  readOnly?: boolean;
}

export function StarRating({ value, onChange, readOnly }: StarRatingProps) {
  const rating = value ?? 0;
  const isUnrated = value == null;
  const isInteractive = !readOnly && onChange;

  const label = isUnrated
    ? "Unrated"
    : `Rating: ${rating} out of 5`;

  function renderStars() {
    const stars: React.ReactNode[] = [];
    for (let i = 1; i <= 5; i++) {
      const isFull = rating >= i;
      const isHalf = !isFull && rating >= i - 0.5;

      const glyph = isFull ? "★" : isHalf ? "½" : "☆";
      const starClass = [
        styles.star,
        isFull || isHalf ? styles.filled : "",
        isUnrated ? styles.disabled : "",
      ]
        .filter(Boolean)
        .join(" ");

      if (isInteractive) {
        stars.push(
          <button
            key={i}
            type="button"
            className={`${starClass} ${styles.interactive}`}
            onClick={() => onChange(i)}
            aria-label={`Rate ${i} stars`}
          >
            {glyph}
          </button>,
        );
      } else {
        stars.push(
          <span key={i} className={starClass}>
            {glyph}
          </span>,
        );
      }
    }
    return stars;
  }

  return (
    <div className={styles.container} aria-label={label}>
      {renderStars()}
    </div>
  );
}
```

```tsx
  const className = [
    styles.pill,
    isOffFlavor ? styles.offFlavor : "",
    selected ? styles.selected : "",
    onClick ? styles.clickable : "",
  ]
    .filter(Boolean)
    .join(" ");

  return (
    <span
      className={className}
      style={pillStyle}
      onClick={onClick}
      title={name}
    >
      <span className={styles.dot} style={{ background: color }} />
      {selected && <span className={styles.check}>✓</span>}
      <span className={styles.name}>{name}</span>
      {onRemove && (
        <button
          type="button"
          className={styles.remove}
          onClick={(e) => {
            e.stopPropagation();
            onRemove();
          }}
          aria-label={`Remove ${name}`}
        >
          ✕
        </button>
      )}
    </span>
  );
}

function hexToRgb(hex: string): string {
  const r = parseInt(hex.slice(1, 3), 16);
  const g = parseInt(hex.slice(3, 5), 16);
  const b = parseInt(hex.slice(5, 7), 16);
  return `${r},${g},${b}`;
}
```

```tsx
import { useState } from "react";
import { useQuery, useMutation } from "@apollo/client/react";
import { useNavigate } from "react-router-dom";
import { MY_ROASTS_QUERY, MY_BEANS_QUERY, UPDATE_ROAST_RATING } from "../graphql/operations";
import { StarRating } from "../components/StarRating";
import { FlavorPill } from "../components/FlavorPill";
import { formatDuration, formatTemp, formatDate } from "../lib/formatters";
import styles from "./DashboardPage.module.css";

export function DashboardPage() {
  const navigate = useNavigate();
  const { data, loading, error } = useQuery(MY_ROASTS_QUERY);
  const { data: beansData } = useQuery(MY_BEANS_QUERY);
  const [updateRating] = useMutation(UPDATE_ROAST_RATING);

  const [search, setSearch] = useState("");
  const [beanFilter, setBeanFilter] = useState("");
  const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set());

  if (loading) return <div className={styles.loading}>Loading roasts...</div>;
  if (error) return <div className={styles.error}>Error loading roasts</div>;

  const roasts = data?.myRoasts ?? [];
  const beans = beansData?.myBeans ?? [];

  const filtered = roasts.filter((r) => {
    const matchesSearch =
      !search ||
      r.bean.name.toLowerCase().includes(search.toLowerCase()) ||
      r.notes?.toLowerCase().includes(search.toLowerCase());
    const matchesBean = !beanFilter || r.bean.id === beanFilter;
    return matchesSearch && matchesBean;
  });

  function handleRating(roastId: string, rating: number) {
    updateRating({ variables: { id: roastId, input: { rating } } });
  }

  function toggleSelect(id: string) {
    setSelectedIds((prev) => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  }

  function devDeltaT(r: typeof roasts[number]): number | null {
    if (r.roastEndTemp != null && r.firstCrackTemp != null) {
      return r.roastEndTemp - r.firstCrackTemp;
    }
    return null;
  }

  const userBean = (beanId: string) =>
    beans.find((ub) => ub.bean.id === beanId);

  if (roasts.length === 0) {
    return (
      <div className={styles.empty}>
        <h2>No roasts yet</h2>
        <p>Upload your first .klog file to get started.</p>
      </div>
    );
  }

  return (
    <div className={styles.page}>
      <div className={styles.toolbar}>
        <div>
          <h1 className={styles.title}>My Roasts</h1>
          <p className={styles.subtitle}>
            {roasts.length} roast{roasts.length !== 1 ? "s" : ""} ·{" "}
            {beans.length} bean{beans.length !== 1 ? "s" : ""}
          </p>
        </div>
        <div className={styles.filters}>
          <input
            type="text"
            placeholder="Search bean or notes..."
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            className={styles.searchInput}

          <select
            value={beanFilter}
            onChange={(e) => setBeanFilter(e.target.value)}
            className={styles.beanSelect}
          >
            <option value="">All beans</option>
            {beans.map((ub) => (
              <option key={ub.bean.id} value={ub.bean.id}>
                {ub.shortName || ub.bean.name}
              </option>
            ))}
          </select>
        </div>
      </div>

      <div className={styles.tableHeader}>
        <span className={styles.colBean}>Bean</span>
        <span className={styles.colDate}>Date</span>
        <span className={`${styles.colNotes} ${styles.desktopOnly}`}>Notes</span>
        <span className={`${styles.colFlavors} ${styles.desktopOnly}`}>Flavors</span>
        <span className={`${styles.colNotesFlavors} ${styles.mobileOnly}`}>Notes & Flavors</span>
        <span className={styles.colDev}>Dev Time</span>
        <span className={styles.colDevDt}>Dev ΔT</span>
        <span className={styles.colRating}>Rating</span>
      </div>

      {filtered.map((roast) => {
        const ub = userBean(roast.bean.id);
        const allFlavors = [...(roast.flavors ?? []), ...(roast.offFlavors ?? [])];
        const delta = devDeltaT(roast);

        return (
          <div
            key={roast.id}
            className={styles.row}
            onClick={() => navigate(`/roasts/${roast.id}`)}
          >
            <div className={styles.colBean}>
              <div className={styles.beanName}>{roast.bean.name}</div>
              {ub?.shortName && (
                <div className={styles.shortName}>{ub.shortName}</div>
              )}
            </div>
            <div className={styles.colDate}>
              {formatDate(roast.roastDate)}
            </div>
            <div className={`${styles.colNotes} ${styles.desktopOnly}`}>
              {roast.notes && (
                <span className={styles.notesText}>{roast.notes}</span>
              )}
            </div>
            <div className={`${styles.colFlavors} ${styles.desktopOnly}`}>
              <div className={styles.pillRow}>
                {allFlavors.slice(0, 3).map((f) => (
                  <FlavorPill
                    key={f.id}
                    name={f.name}
                    color={f.color}
                    isOffFlavor={f.isOffFlavor}
                  />
                ))}
                {allFlavors.length > 3 && (
                  <span className={styles.moreChip}>
                    +{allFlavors.length - 3}
                  </span>
                )}
              </div>
            </div>
            <div className={`${styles.colNotesFlavors} ${styles.mobileOnly}`}>
              {roast.notes && (
                <span className={styles.notesText}>{roast.notes}</span>
              )}
              <div className={styles.pillRow}>
                {allFlavors.slice(0, 3).map((f) => (
                  <FlavorPill
                    key={f.id}
                    name={f.name}
                    color={f.color}
                    isOffFlavor={f.isOffFlavor}
                  />
                ))}
              </div>
            </div>
            <div className={styles.colDev}>
              {formatDuration(roast.developmentTime)}
            </div>
            <div className={styles.colDevDt}>
              {delta != null ? formatTemp(delta, "CELSIUS") : "—"}
            </div>
            <div
              className={styles.colRating}
              onClick={(e) => e.stopPropagation()}
            >
              <StarRating
                value={roast.rating ?? null}
                onChange={(r) => handleRating(roast.id, r)}
              />
            </div>
          </div>

      {selectedIds.size >= 2 && (
        <div className={styles.compareBar}>

            className={styles.compareButton}

          >
            Compare {selectedIds.size} roasts

        </div>

    </div>
  );
}
```

```ts
import { GraphQLError } from "graphql";

interface BeanScrapeResult {
  name: string | null;
  origin: string | null;
  process: string | null;
  elevation: string | null;
  bagNotes: string | null;
  suggestedFlavors: string[];
}

export class ScrapingService {
  async scrapeBeanUrl(url: string): Promise<BeanScrapeResult> {
    // Validate URL is from Sweet Maria's
    if (!url.includes("sweetmarias.com")) {
      throw new GraphQLError("Only Sweet Maria's URLs are supported in v1", {
        extensions: { code: "BAD_USER_INPUT" },
      });
    }

    const response = await fetch(url, {
      headers: { "User-Agent": "CoffeeRoastTracker/1.0" },
    });
    if (!response.ok) {
      throw new GraphQLError(`Failed to fetch URL: ${response.status}`, {
        extensions: { code: "BAD_USER_INPUT" },
      });
    }

    const html = await response.text();
    return this.parseProductPage(html);
  }

  parseProductPage(html: string): BeanScrapeResult {
    // Extract data using regex patterns matched to Sweet Maria's HTML structure.
    // This is intentionally simple — no DOM parser dependency.
    const name = this.extractMatch(html, /<h1[^>]*class="product_title"[^>]*>(.*?)<\/h1>/s);
    const origin = this.extractTableField(html, "Region");
    const process = this.extractTableField(html, "Processing");
    const elevation = this.extractTableField(html, "Elevation");
    const bagNotes = this.extractMatch(html, /<div[^>]*class="farm-notes"[^>]*>\s*<p>(.*?)<\/p>/s);

    // Extract flavor descriptors from cupping notes if present
    const suggestedFlavors: string[] = [];
    const cuppingNotes = this.extractMatch(html, /cupping[- ]?notes[^>]*>(.*?)</si);
    if (cuppingNotes) {
      suggestedFlavors.push(
        ...cuppingNotes.split(/[,;]/).map((s) => s.trim()).filter(Boolean),
      );
    }

    return {
      name: name?.trim() ?? null,
      origin: origin?.trim() ?? null,
      process: process?.trim() ?? null,
      elevation: elevation?.trim() ?? null,
      bagNotes: bagNotes?.trim() ?? null,
      suggestedFlavors,
    };
  }

  private extractMatch(html: string, pattern: RegExp): string | null {
    const match = html.match(pattern);
    return match?.[1]?.replace(/<[^>]*>/g, "") ?? null;
  }

  private extractTableField(html: string, label: string): string | null {
    const pattern = new RegExp(
      `<td[^>]*>${label}</td>\\s*<td[^>]*>(.*?)</td>`,
      "si",
    );
    return this.extractMatch(html, pattern);
  }
}
```
